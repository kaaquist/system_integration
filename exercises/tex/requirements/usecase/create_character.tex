%Scope: the system under design
%\addfield{Scope:}{System-wide}

\addfield{Description:}{The player creates a new character on the server}

%Primary Actor: Calls on the system to deliver its services.
\addfield{Actor:}{Player}

%Stakeholders and Interests: Who cares about this use case and what do they want?
%\additemizedfield{Stakeholders and Interests:}{

%Preconditions: What must be true on start and worth telling the reader?
\addfield{Preconditions:}{}
%when multiple
%\additemizedfield{Preconditions:}{} 

%Main Success Scenario: A typical, unconditional happy path scenario of success.
\addscenario{Main Scenario:}{
	\item The player provides the server with information about what character name, e-mail address and password he wishes to use on the server. 
	\item The player presses the ''Create Character'' button.
}

%Extensions: Alternate scenarios of success or failure.
\addscenario{Alternative Scenario:}{
	\item[2.a] Character name already in use:
		\begin{enumerate}
		\item[1.] System shows a failure message, explaining the issue
		\item[2.] The player retries step 1 with a new character name
		\end{enumerate}
	\item[2.b] Invalid e-mail address:
		\begin{enumerate}
		\item[1.] System shows failure message, explaining the issue
		\item[2.] Player returns to step 2 and corrects the e-mail address
		\end{enumerate}
	\item[2.c] Password doesn't comply with the rules set up on the system
		\begin{enumerate}
		\item[1.] System shows a failure message, explaining the issue and what the password has to follow
		\item[2.] The player retries step 1 with a new password, following the rules
		\end{enumerate}
}
